ダブリング
ダブリングとはなんですか -> 2^(k-1) 個先のさらに2^(k-1) 個先は 2^k 個先という事実、これにつきます。
ダブリングはそこまでやることないのですが、ブレインフォグで何も考えられなくなってしまった時のために一応コードを残しておこうという所存でございます。


  1. #include<iostream>
  2. #include<vector>
  4. const long long INF = 1e18;
  5. using namespace std;
  7. /*
  8.     A = {A_1,A_2, ... , A_N}があります。
  9.     i は A_iに一方通行です。
  10.     最初1にいるとき、K(K<10^18)回道を移動した後あなたはどこにいますか。
  11. */
  15. int main(){
  16.     int N ;long long K ;
  17.     cin >> N >> K;
  18.     vector<vector<int>> dp(65 , vector<int>(N+1));
  19.     vector<int> A(N);
  20.     for(int i = 0; i<N;i++)cin >> A[i];
  22.     for(int i = 1 ; i <= N ; i++){
  23.         dp[0][i] = A[i-1];
  24.     }
  26.     for(int k = 1 ; k < 65 ; k++){
  27.         for(int x = 1 ; x<=N ; x++){
  28.             dp[k][x] = dp[k-1][dp[k-1][x]];//2^個先は、2^(k-1)個先の2^(k-1)個先
  29.         }
  30.     }
  31.     int now = 1;
  33.     for(int bit = 0 ; bit < 64 ; bit++){
  34.         if((K & (1ll<<bit))!=0){
  35.             now = dp[bit][now];
  36.         }
  37.     }
  39.     cout << now << endl;
  42.     return 0;
  43. }
