先日書いたSplay木についていろいろ考えているときに思いついたクエリ処理を問題にしてみました(2)。

問題:SearchableArray
Q個のクエリが与えられるので、処理してください。
なお、クエリの種類は t によって指定される。
  始め、空の数列:Aがある。
  クエリ１: t=1の時、数列Aの i (0-index) 番目の位置に X を挿入する。(push)
  クエリ２: t=2の時、Aの i (0-index) 番目を削除する。(pop)
  クエリ３: t=3の時、Aに存在する X の位置を全て出力する

制約:0 < Q <= 500000 , t = {1,2,3} , i が 存在しない位置を指し示すことはない(ただし、pushで末尾を指定することはある)

入力: 一行目に整数Q , 続くQ行にはクエリが与えられる。各行の初めの入力はtで、クエリの形式は以下のいずれかである。
  
  t = 1 -> "1 i x"
  t = 2 -> "2 i"
  t = 3 -> "3 x"


挿入、削除は平衡二分木を使うとO(logN)で実装できます。

ここでおさらいです。
SplayTreeの実装は、各頂点の情報を持たせる構造体(のポインタ):Node 同士が、お互いがどの頂点と繋がっているのかの情報を持つことで木の構造を保っていました。
どのように頂点をつなげるかというと、あるNodeには、自分の左右の子、上の親、それぞれのNodeのポインタを持たせることで、コピーではない実際の頂点(を表すデータ構造)につなげていました。

ここで、ポインタについてもおさらいです。
具体例を使って説明します。
    あなたは大事な情報 S を変数として持っています。
    S は関数 F のパラメータとして使い、Fの計算過程で S も整形(情報が書き換えられる)されます。
    F は非常に複雑な計算をするため、S の変化を追うことはできません。
こういう時、C++ならFの引数を参照渡しするか、そもそも、Sをポインタで持っておくといいです。
変数をポインタで持っておくと、そのポインタの中身が改変された時、そのポインタを持つ変数全てについて、その指し示すポインタの中身が変わります。


ここで、クエリ 3 は次のように実現できます。
まず標準ライブラリの std::map<int , vector<Node*> > p; を用意して、
Aのi番目にxを追加する際は、Aを表す SplayTree に Node(値 = x) のポインタ *N を追加すると共に、p[x] に *N を追加します。

こうして、p[x] := 値としてxを持つSplayTreeの頂点たちを保管したvectorとなりました。

ここで、Aに対して新たにpushやpopを繰り返すと、Aを表すSplayTreeの構造が変わり、それに合わせて各頂点の位置関係も変わります。
しかし、pに追加した頂点のポインタと、実際にSplayTreeを構築する頂点のポインタは同じものを使っているので、SplayTreeの頂点情報が変化すると同時に、pの中身の頂点も持つ情報を変化させています。
よって、pushやpopを繰り返しても、pの中身も連動して変化するため、pの中のNodeの情報からそれぞれのNodeがSplayTreeの中でどの位置にあるかを計算することができます。
ここで、位置とはSplayTreeの頂点のうち、左から何番目かをさします。

あるNodeのポインタが、SplayTree内のどの位置にあるかは、その頂点の親を辿っていけばわかります。
自分の親ノードから見て、自分が右の子ノードであれば、親の左の子からなる部分木は、全て自分よりも左にあるということを利用して、

今いる頂点をnow , 自分の親ノードをparent, 整数:cnt = 0;とすると、
(1)nowがparentの右の子ノードの場合、cntに parent の左の子ノードの部分木のサイズ + 1 を足す
(2)nowがparentの左の子ノードの場合、何もしない
(3)now を parentにする
(1) , (2) , (3) の操作を、now が SplayTreeのルート(根)に到達するまで繰り返す。
すると、最終的に求まった cnt がSplayTree内での位置となる。これで、pの中のNode*から、直接位置を計算することができました。

ただし、popがあった場合だと、popされたNode*はAには存在しないのに対して、pから効率よくpopする手段がないので、pの中身にアクセスした時にすでにpopされたものかどうかを判定しなければなりません。
また、popがあると、popした要素に対応するNode*をnullptrにする。

判定方法は以下の3つです。
1 : p[x]の中身のNode* から直接計算して得られた位置を cnt とすると、A[cnt]にアクセスして値が合致するかを判定する
2 : cntが同じ値を二度以上とった場合、列挙しない
3 : p[x]の中身がnullptrなら、p[x]から削除する

  515.     */
  516.     SplayNode *insert_sub(int index , SplayNode *NODE , SplayNode *root){
  517.         if(isExist(root) == false){
  518.             pos[NODE->Value].push_back(NODE);
  519.             root=NODE;
  520.             Size=1;
  521.             return root;
  522.         }
  523.         if(Size<index || index<0)return root;
  525.         pos[NODE->Value].push_back(NODE);
  526.         std::pair<SplayNode*,SplayNode*> Trees = split(index,root);
  528.         SplayNode *leftRoot = Trees.first;
  529.         SplayNode *rightRoot = Trees.second;
  530.         Size++;
  532.         return merge(merge(leftRoot , NODE),rightRoot);
  533.     }
  543.     /*
  544.         rootを根とする部分木において、
  545.         左からindexの場所を削除して、両端をくっつける
  546.         firstは消したやつ以外をmergeした時のroot
  547.         secondは消したSplayNode
  548.     */
  549.     std::pair<SplayNode*,SplayNode*> Delete_sub(int index , SplayNode *root){
  550.         
  551.         if(isExist(root) == false){
  552.             return std::make_pair(root,root);
  553.         }
  554.         if(index<0 || index >= Size){
  555.             return std::make_pair(root,root);
  556.         }
  557.         
  558.         if(Size == 1){
  559.             Size = 0;
  560.             return std::make_pair(nullptr,nullptr);
  561.         }
  563.         root = getNode(index,root);
  564.         
  566.         SplayNode *leftRoot = root->left;
  567.         SplayNode *rightRoot = root->right;
  569.         if(isExist(leftRoot))leftRoot->parent = nullptr;
  570.         if(isExist(rightRoot))rightRoot->parent = nullptr;
  571.         root->left = nullptr;
  572.         root->right = nullptr;
  574.         Size--;
  575.         
  576.         root->update();
  577.         
  578.         return std::make_pair(merge(leftRoot,rightRoot) , root );
  579.     }
  588.     /*
  589.         rootを根とする部分きにおいて、
  590.         0-indexの区間[l,r)を右シフト(右端は左端へ)する
  591.     */
  592.     SplayNode *shift_right_sub(int l , int r , SplayNode *root ){
  593.         r--;//半開区間なので
  595.         std::pair<SplayNode*,SplayNode*> tmp = Delete_sub(r,root);
  596.         root = tmp.first;
  597.         SplayNode *N = tmp.second;
  599.         return insert_sub(l,N,root);
  601.     }
  605.     /*
  606.         rootを根とする部分きにおいて、
  607.         0-indexの区間[l,r)を左シフト(左端は右端へ)する
  608.     */
  609.     SplayNode *shift_left_sub(int l , int r , SplayNode *root ){
  610.         r--;//半開区間なので
  612.         std::pair<SplayNode*,SplayNode*> tmp = Delete_sub(l,root);
  613.         root = tmp.first;
  614.         SplayNode *N = tmp.second;
  616.         return insert_sub(r,N,root);
  618.     }
  628.     /*
  629.         rootを根とする部分木の[l,r)の区間取得クエリ
  630.         modeが 0->Min, 1->Max ,2->Sum
  631.     */
  632.     std::pair<SplayNode*,T> RangeQuery_sub(int l , int r , int mode , SplayNode *root){
  634.         SplayNode *leftRoot , *centerRoot , *rightRoot;
  635.         r--;//半開区間なので
  637.         std::pair<SplayNode*,SplayNode*> tmp = split(r+1,root);
  638.         rightRoot = tmp.second;
  639.         
  640.         tmp = split(l,tmp.first);
  641.         leftRoot = tmp.first;
  642.         centerRoot = tmp.second;
  643.         
  644.         T ans;
  645.         //modeによって取得する値が違う
  646.         if(mode == 0){
  647.             ans = centerRoot->Min;
  648.         }else if(mode == 1){
  649.             ans = centerRoot->Max;
  650.         }else if(mode == 2){
  651.             ans = centerRoot->Sum;
  652.         }else{
  653.             ans = centerRoot->Sum;
  654.         }
  656.         return std::make_pair(merge(merge(leftRoot,centerRoot),rightRoot),ans);
  657.     }

  889.     //数列内の、xの位置を全てvector<int>に入れて返す
  890.     vector<int> get_position(T x){
  891.         if(Size == 0)return vector<int>(0);
  893.         vector<SplayNode*> pos_x = pos[x];
  894.         vector<int> res;
  896.         int size = pos_x.size();
  898.         
  899.         //shift_rightなどで、辺や頂点を切ったり張ったりすると,
  900.         //pos[x]の中で同じ頂点が重複して存在することになるので、そこも弾くためにチェック
  901.         map<int,bool> checked;
  903.         for(int i = size -1 ; i >= 0 ; i--){
  904.             SplayNode *Nd = pos_x[i];
  905.             
  906.  
  907.             
  908.             if(isExist(Nd) == false)pos_x.erase(pos_x.begin()+i);//削除済みを弾く
  909.             else if(Nd->Value != x)pos_x.erase(pos_x.begin()+i);//値の変更ずみを弾く
  910.             else{
  912.                 //NODEのindex関数で、上の辺を辿って、自分がどの位置にあるのかを取得(splayして自分を上に持って行ってもいいけど、Deleteで木から削除された頂点を弾く場合の処理が面倒)
  913.                 int index = Nd->index();
  914.             
  915.                 /*
  916.                     また、木からDeleteされた頂点を弾くと書いたが、shiftなどの操作によって、一度は木から削除されたが、後に木に戻ってきた頂点は弾かないようにしている
  917.                 */
  918.                 T val = get(index).Value;
  920.                 //index番目が、本当にxと同じかどうかをチェック(deleteによって無効になったノードもあるので)
  921.                 if(val == x){
  922.                     //すでにみていた場合、このNODEはshiftなどの操作により、pos[x]に複数存在するということなので、pop
  923.                     if(checked[index])pos_x.erase(pos_x.begin()+i);
  924.                     else res.push_back(index);
  926.                     checked[index] = true;
  927.                 }else pos_x.erase(pos_x.begin()+i);
  928.                 
  929.                 
  930.             }
  931.         }
  932.         pos[x] = pos_x;//削除とかをしたので、更新しておく
  933.         
  934.         sort(res.begin() , res.end());
  935.         return res;
  936.     }
