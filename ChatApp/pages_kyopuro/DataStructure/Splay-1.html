平衡二分探索木(1)
今日は平衡二分探索木を勉強しました。
Splay木というらしいです。
かつっぱさん（youtubeで調べてみて）の動画を見て一応大雑把に理解はできたのですが、大事な部分がほぼ同じコードになってしまったので、いつか完全に空で描けるようになりたいです。
しかし、バグり散らかしたり、コードの意味を読み取ったりでとんでもない時間を費やしてしまいました。。。
恐らくアルゴリズムの理解が間違っているところや、私の実装ではうまくいかないような状況もあると思いますが、どうか温かい目でお願いします。


SplayTreeにはtemplateを渡すようにしていますので、頂点に複数個の値を持たせたい時は、自作でstructを作るといいです。
いちいち各所に現れる演算子や関数( min ,max , + や > など )を書き直すのは面倒なので、T に struct を渡してそのstructに必要な演算子を乗せたほうが楽です。

  35.     関数の引数:rootに何も入れなかった場合、全体の根であるRootが自動で代入される
  37.     Treeの上の操作は、Glaphのデータ構造をを用意して行うのではなく、

  41.     rootを頂点とする部分木において{

  44.         merge(l_root , r_root) := 頂点がl_root ,r_rootであるような部分木を連結

  51.     これらを駆使し、様々な操作を実装可能

  55. template<class T>
  56. class SplayTree{
  57.     public:
  60.     /*
  61.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  62.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  63.             SplayTreeの、Nodeのデータ構造。
  64.             SplayTreeでは、Glaphの配列上で操作のやり取りをするのではなく、
  65.             SplayNodeというデータ構造に持たせたデータでやり取りをする
  66.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  67.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  69.     */
  71.     struct SplayNode{
  73.         SplayNode *parent;//親ノード
  74.         SplayNode *left;//左の子ノード
  75.         SplayNode *right;//右の子ノード
  77.         T Value;//値(普通、頂点には値や重みが割り当てられることが多い)
  78.         T Min,Max,Sum;//部分木のうち、値の最大、最小、和
  80.         int SubTeeSize = 1;//1は初期化値(自分自身)
  83.         SplayNode(){
  84.             parent = nullptr;
  85.             left = nullptr;
  86.             right = nullptr;
  87.             
  88.         }
  89.         SplayNode(T val){
  90.             parent = nullptr;
  91.             left = nullptr;
  92.             right = nullptr;
  93.             Value = val;
  94.             Min = val;
  95.             Max = val;
  96.             Sum = val;
  97.         }
  100.         bool isExist(SplayNode *Nd){
  101.             if(Nd!=nullptr)return true;
  102.             else return false;
  103.         }
  107.         /*
  108.             ある地点を回転させる
  109.         */
  110.         void rotate(){
  112.             SplayNode *Parent , *GrandParent , *Child;
  113.             Parent = this->parent;
  114.             GrandParent = Parent->parent;
  116.             if(isExist(Parent) == false){
  117.                 return;
  118.             }
  122.             //元の親に対する位置で場合分け
  123.             if(Parent->left == this){
  124.                 Child = this->right;//Childはnullptrかもしれない
  125.                 this->right = Parent;
  126.                 Parent->left = Child;
  128.             }else if(Parent->right == this){
  129.                 Child = this->left;
  130.                 this->left = Parent;
  131.                 Parent->right = Child;
  133.             }
  135.             if(isExist(GrandParent)){
  136.                 //GrandParentのどちらの辺なのかで場合分け
  137.                 if(Parent == GrandParent->left){
  138.                     GrandParent->left = this;
  140.                 }else if(Parent == GrandParent->right){
  141.                     GrandParent->right = this;
  142.                     
  143.                 }
  144.             }
  146.             //GrandParentがnullptrの場合でも代入OK
  147.             this->parent = GrandParent;
  150.             //親と位置を入れ替えるので
  151.             Parent->parent = this;
  155.             //Childは移動するので。
  156.             if(isExist(Child))Child->parent = Parent;
  159.             /*
  160.                 Nodeの持つ情報をupdateしないといけないが、順番に注意！！
  161.                 ChildはChild以下の部分木が変化してないので何もしなくてOK
  162.                 GrandParentは、部分木の中の順番が変わったけど、集合としては変わってないので何もしなくてOK
  163.                 Parentと自分自身(this)は順番が変わって、部分木の内容も変わったので、
  164.                 下から　Parent　? this の順番でupdateする
  165.             */
  166.             Parent->update();
  167.             this->update();
  169.             
  171.             return;
  173.         }
  181.         /*
  182.             自分の親から見た自分の位置
  183.             0 -> 親がそもそもいない or 例外の場合
  184.             1 -> 親から見て左の子の場合
  185.             2 -> 親から見て右の子の場合
  186.         */
  187.         int state(){
  188.             if(isExist(this->parent)== false){
  189.                 return 0;
  190.             }else{
  191.                 if(this->parent->left == this){
  192.                     return 1;
  193.                 }else if(this->parent->right == this){
  194.                     return 2;
  195.                 }else {
  196.                     return 0;
  197.                 }
  198.             }
  199.         }
  207.         /*
  208.             あるNodeを回転を駆使し一番上まで持っていく
  209.             この時、回転する順番に注意
  210.         */
  211.         void splay(){
  212.             while(isExist(this->parent)){
  213.                 if(isExist(this->parent->parent)==false){
  214.                     //(自分のGrandParentがいない場合、rotateは一回のみ)
  215.                     this->rotate();
  216.                     return;
  219.                 //それ以外の場合、(自分、親、親の親)の位置関係が大事
  220.                 }else if(this->state() == this->parent->state()){
  221.                     //GrandParentからの道が(右、右)or (左、左)の場合
  222.                     //GrandParentを一つ下に下ろして、自分を上げる
  223.                     this->parent->rotate();
  224.                     this->rotate();
  226.                 }else{
  227.                     //それ以外の場合、単純に自分を2回上に上げれば良い
  228.                     this->rotate();
  229.                     this->rotate();
  230.                 }
  231.             }
  235.             return;
  236.         }
  243.         /*
  244.             サイズなど、Nodeの持っている情報をupdateする(順番大事)
  245.         */
  246.         void update(){
  247.             this->SubTeeSize = 1;
  248.             this->Min = this->Value;
  249.             this->Max = this->Value;
  250.             this->Sum = this->Value;
  253.             if(isExist(this->left)){
  254.                 this->SubTeeSize += this->left->SubTeeSize;
  255.                 if(this->left->Min < this->Min)this->Min = this->left->Min;
  256.                 if(this->left->Max > this->Max)this->Max = this->left->Max;
  257.                 this->Sum += this->left->Sum;
  258.             }
  260.             if(isExist(this->right)){
  261.                 this->SubTeeSize += this->right->SubTeeSize;
  262.                 if(this->right->Min < this->Min)this->Min = this->right->Min;
  263.                 if(this->right->Max > this->Max)this->Max = this->right->Max;
  264.                 this->Sum += this->right->Sum;
  266.             }
  268.             return;
  269.         }
  271.     };
  275.     /*
  276.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  277.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  278.             ここからやっとSplayTreeに載せる関数たちの出番
  279.             rootを根とする部分木において、
  280.               split(ind,root) , merge(l_root, r_root) , delete_sub(ind,root) , insert_sub()　など
  281.             これらは主にclass内で使われる。
  282.             classの外で使う関数として、
  283.                 Delete , insert, RangeSumQuery , shiftなどが想定される
  284.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  285.         ------------------------------------------------------------------------------------------------------------------------------------------------------------
  287.     */
  291.     //これは、Nodeを管理する配列のサイズで、木のサイズではない(変更しないので、あらかじめ大きめにとっておく)
  292.     int Size = 0;
  293.     //この木のroot,splitした後はmergeして戻す！！rootが迷子にならないように！！
  294.     SplayNode *Root = nullptr;
  299.     SplayTree(){
  301.     }
  303.     SplayTree(int size_){
  304.   
  305.     }
  307.     bool isExist(SplayNode *Nd){
  308.         if(Nd==nullptr){
  309.             return false;
  310.         }else{
  311.             return true;
  312.         }
  313.     }
  318.     /*
  319.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
  320.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
  321.         ここからは、クラス内で動かす関数たちです。俗にいうブラックボックスです。
  322.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
  323.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
  324.     */
  327.     private:
  335.     /*
  336.         rootを根とする部分木において、
  337.         indexを左からの0-indexとすると、
  339.         求めたいindexになるまで降りていく
  341.     */
  342.     SplayNode *getNode(int index , SplayNode *root){
  343.         
  344.         if(isExist(root)==false)return root;
  345.         
  347.         SplayNode *now = root;
  349.         while(1){
  351.             int left_size = 0;
  352.             if(isExist(now->left)){
  353.                 left_size = now->left->SubTeeSize;
  354.             }
  356.             if(index < left_size){
  357.                 now = now->left;
  358.             }else if(index == left_size){
  359.                 now->splay();
  360.                 break;
  361.             }else if(index > left_size){
  362.                 now = now->right;
  363.                 index = index - left_size-1;
  364.             }
  366.         }
  369.         return now;
  370.     }
  384.     /*
  385.         それぞれ、leftrootとrightrootが根であるような部分木をくっつける
  387.         splitした後、これをしてまたsplitしたものをくっつける
  388.         splitしたまま放置すると分割された木のrootが行方不明になりそう
  389.     */
  390.     SplayNode *merge(SplayNode *leftRoot , SplayNode *rightRoot){
  392.        
  393.         if(isExist(leftRoot) == false)return rightRoot;//右のみあり
  394.         if(isExist(rightRoot) == false )return leftRoot;//左のみあり
  397.         leftRoot = getNode(leftRoot->SubTeeSize-1 , leftRoot);
  398.         leftRoot->right = rightRoot;
  399.         rightRoot->parent = leftRoot;
  400.         leftRoot->update();
  401.         
  402.         return leftRoot;
  403.     }
  413.     /*
  414.          rootを根とする部分木において、
  415.         左からleftnum番目の頂点で左右にsplit。
  416.         splitした後はmergeしないと,splitした木の頂点が行方不明になりそう
  417.     */
  418.     std::pair<SplayNode*,SplayNode*> split(int leftnum, SplayNode *root){
  420.         if(leftnum<=0)return std::make_pair(nullptr , root);
  422.         if(leftnum >= root->SubTeeSize)return std::make_pair(root, nullptr);
  426.         root = getNode(leftnum , root);
  427.         
  428.         SplayNode *leftRoot , *rightRoot;
  430.         leftRoot = root->left;
  431.         
  432.         rightRoot = root;
  434.         
  435.         if(isExist(rightRoot))rightRoot->left = nullptr;
  436.         
  437.         if(isExist(leftRoot))leftRoot->parent = nullptr;
  438.         
  439.         rightRoot->update();
  442.         return std::make_pair(leftRoot,rightRoot);
  444.     }
  450.     /*
  451.          rootを根とする部分木において、
  452.         indexの場所を切り離して、NODEを追加して元に戻す(merge)
  453.     */
  454.     SplayNode *insert_sub(int index , SplayNode *NODE , SplayNode *root){
  455.         if(isExist(root) == false){
  456.             root=NODE;
  457.             Size=1;
  458.             return root;
  459.         }
  460.         if(Size<index || index<0)return root;
  462.         std::pair<SplayNode*,SplayNode*> Trees = split(index,root);
  464.         SplayNode *leftRoot = Trees.first;
  465.         SplayNode *rightRoot = Trees.second;
  466.         Size++;
  468.         return merge(merge(leftRoot , NODE),rightRoot);
  469.     }
  478.     /*
  479.         rootを根とする部分木において、
  480.         左からindexの場所を削除して、両端をくっつける
  481.     */
  482.     std::pair<SplayNode*,SplayNode*> Delete_sub(int index , SplayNode *root){
  483.         
  484.         if(isExist(root) == false){
  485.             
  486.             return std::make_pair(root,root);
  487.         }
  488.         if(index<0 || index >= Size){
  490.             return std::make_pair(root,root);
  491.         }
  492.         
  493.         if(Size == 1){
  495.             Size = 0;
  496.             root = nullptr;
  497.             return std::make_pair(nullptr,nullptr);
  498.         }
  500.         root = getNode(index,root);
  502.         SplayNode *leftRoot = root->left;
  503.         SplayNode *rightRoot = root->right;
  505.         if(isExist(leftRoot))leftRoot->parent = nullptr;
  506.         if(isExist(rightRoot))rightRoot->parent = nullptr;
  507.         root->left = nullptr;
  508.         root->right = nullptr;
  510.         Size--;
  511.         
  512.         root->update();
  513.         
  514.         return std::make_pair(merge(leftRoot,rightRoot) , root );
  515.     }
  524.     /*
  525.         rootを根とする部分きにおいて、
  526.         0-indexの区間[l,r)を右シフト(右端は左端へ)する
  527.     */
  528.     SplayNode *shift_right_sub(int l , int r , SplayNode *root ){
  529.         r--;//半開区間なので
  531.         std::pair<SplayNode*,SplayNode*> tmp = Delete_sub(r,root);
  532.         root = tmp.first;
  533.         SplayNode *N = tmp.second;
  535.         return insert_sub(l,N,root);
  537.     }
  541.     /*
  542.         rootを根とする部分きにおいて、
  543.         0-indexの区間[l,r)を左シフト(左端は右端へ)する
  544.     */
  545.     SplayNode *shift_left_sub(int l , int r , SplayNode *root ){
  546.         r--;//半開区間なので
  548.         std::pair<SplayNode*,SplayNode*> tmp = Delete_sub(l,root);
  549.         root = tmp.first;
  550.         SplayNode *N = tmp.second;
  552.         return insert_sub(r,N,root);
  554.     }
  564.     /*
  565.         rootを根とする部分木の[l,r)の区間取得クエリ
  566.         modeが 0->Min, 1->Max ,2->Sum
  567.     */
  568.     std::pair<SplayNode*,T> RangeQuery_sub(int l , int r , int mode , SplayNode *root){
  570.         SplayNode *leftRoot , *centerRoot , *rightRoot;
  571.         r--;//半開区間なので
  573.         std::pair<SplayNode*,SplayNode*> tmp = split(r+1,root);
  574.         rightRoot = tmp.second;
  575.         
  576.         tmp = split(l,tmp.first);
  577.         leftRoot = tmp.first;
  578.         centerRoot = tmp.second;
  580.         int ans;
  581.         //modeによって取得する値が違う
  582.         if(mode == 0){
  583.             ans = centerRoot->Min;
  584.         }else if(mode == 1){
  585.             ans = centerRoot->Max;
  586.         }else if(mode == 2){
  587.             ans = centerRoot->Sum;
  588.         }else{
  589.             ans = centerRoot->Sum;
  590.         }
  592.         return std::make_pair(merge(merge(leftRoot,centerRoot),rightRoot),ans);
  593.     }
  596.     /*
  597.         SubTreeSize用
  598.     */
  599.     std::pair<SplayNode*,long long> RangeSubTreeSizeQuery_sub(int l , int r , SplayNode *root){
  601.         SplayNode *leftRoot , *centerRoot , *rightRoot;
  602.         r--;//半開区間なので
  604.         std::pair<SplayNode*,SplayNode*> tmp = split(r+1,root);

  606.         
  607.         tmp = split(l,tmp.first);
  608.         leftRoot = tmp.first;
  609.         centerRoot = tmp.second;
  611.         int ans = centerRoot->SubTeeSize;
  613.         return std::make_pair(merge(merge(leftRoot,centerRoot),rightRoot),ans);
  614.     }

  624.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
  625.         ---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

  633.      public:
  637.     /*
  638.         木全体のindex番目のNode(値渡し)を取得

  640.     SplayNode get(int index){
  641.         if(index<0 || index>=Size)return *Root;
  643.         Root = getNode(index,Root);
  644.         Root->update();
  646.         return *(Root);

  652.     void insert(int index , SplayNode *NODE){
  653.         if(index<0 || index> Size)return;
  655.         Root = insert_sub(index , NODE , Root);
  656.         Root->update();
  658.         return;
  660.     }
  664.     /*
  665.         木全体の[l,r)のRMQ
  666.     */
  667.     T RangeMinQuery(int l , int r){
  668.         l = max(0,l);
  669.         r = min(Size,r);
  670.         if(l>=r)return INF;
  672.         std::pair<SplayNode*,long long> tmp = RangeQuery_sub(l,r,0,Root);
  673.         Root = tmp.first;
  674.         Root->update();
  676.         return tmp.second;
  677.     }
  680.     /*
  681.         木全体の[l,r)のRMQ
  682.     */
  683.     T RangeMaxQuery(int l , int r){
  684.         l = max(0,l);
  685.         r = min(Size,r);
  686.         if(l>=r)return NINF;
  688.         std::pair<SplayNode*,long long> tmp = RangeQuery_sub(l,r,1,Root);
  689.         Root = tmp.first;
  690.         Root->update();
  692.         return tmp.second;
  693.     }
  697.     /*
  698.         木全体の[l,r)のRSQ
  699.     */
  700.     T RangeSumQuery(int l , int r){
  701.         l = max(0,l);
  702.         r = min(Size,r);
  703.         if(l>=r)return 0;
