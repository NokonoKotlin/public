遅延評価付き平衡二分木
この木は数列タイプのSplay木で、基本的な動作はこちらとほぼ同じです。

大きく違う点は、遅延評価がついているかどうかということです。

ここで、遅延評価とはノードにとどめておいた作用素をそのノードの持つ値(ValueやMin,Sumなど)に作用させ、その後自分の子ノードに伝播させることです。

遅延評価をつけることによって、ノードのupdate前に左右の子ノードの遅延評価を行なったり、回転時に順番に気をつけて遅延評価及びupdate(値の集約)を行う必要が出てきました。
実装もかなり気を使わないといけないので注意してください。

実装の大変さもあって、できることもかなり増えました。


・RangeAdd(l,r,x) : 半開区間[l,r)内の要素全てにx加算します
・RangeUpdate(l,r,x) : 半開区間[l,r)内の要素全てをxにします
・reverese(l,r) : 半開区間[l,r)内の要素の順番を反転させます。

  112.             GrandParent = Parent->parent;
  113.             Child = nullptr;

  116.                 return;
  117.             }

  119.             Parent->eval();//回転を行う前にevaluate
  120.             this->eval();//回転を行う前にevaluate
  123.             //元の親に対する位置で場合分け
  124.             if(Parent->left == this){
  125.                 Child = this->right;//Childはnullptrかもしれない
  126.                 this->right = Parent;
  127.                 Parent->left = Child;
  129.             }else if(Parent->right == this){
  130.                 Child = this->left;

  132.                 Parent->right = Child;
  134.             }

  137.                 //GrandParentのどちらの辺なのかで場合分け
  138.                 if(Parent == GrandParent->left){
  139.                     GrandParent->left = this;
  141.                 }else if(Parent == GrandParent->right){
  142.                     GrandParent->right = this;
  143.                     
  144.                 }
  145.             }
  147.             //GrandParentがnullptrの場合でも代入OK
  148.             this->parent = GrandParent;
  151.             //親と位置を入れ替えるので
  152.             Parent->parent = this;
  156.             //Childは移動するので。
  157.             if(isExist(Child))Child->parent = Parent;
  160.             /*
  161.                 Nodeの持つ情報をupdateしないといけないが、順番に注意！！
  162.                 ChildはChild以下の部分木が変化してないので何もしなくてOK
  163.                 GrandParentは、部分木の中の順番が変わったけど、集合としては変わってないので何もしなくてOK
  164.                 Parentと自分自身(this)は順番が変わって、部分木の内容も変わったので、
  165.                 下から　Parent　? this の順番でupdateする
  167.                 lazy作用については結構ややこしい図を書いて理解しないといけないが、とりあえずは
  168.                 回転を行う前にthisから遅延評価を降ろして、
  169.                 その後Parentの左右の子の遅延評価を降ろし、Parent->thisの順にupdate()　
  170.             */
  172.             
  173.             Parent->update();
  174.             this->update();
  175.             
  177.             
  179.             return;
  181.         }
  189.         /*
  190.             自分の親から見た自分の位置
  191.             0 -> 親がそもそもいない or 例外の場合
  192.             1 -> 親から見て左の子の場合
  193.             2 -> 親から見て右の子の場合
  194.         */
  195.         int state(){
  196.             if(isExist(this->parent)== false){
  197.                 return 0;
  198.             }else{
  199.                 if(this->parent->left == this){
  200.                     return 1;
  201.                 }else if(this->parent->right == this){
  202.                     return 2;
  203.                 }else {
  204.                     return 0;
  205.                 }
  206.             }
  207.         }
  215.         /*
  216.             あるNodeを回転を駆使し一番上まで持っていく
  217.             この時、回転する順番に注意
  218.         */
  219.         void splay(){
  221.             if(isExist(this) == false)return;
  222.             
  223.             while(isExist(this->parent)){
  224.                 if(isExist(this->parent->parent)==false){
  225.                     //(自分のGrandParentがいない場合、rotateは一回のみ)
  226.                     this->rotate();
  227.                     return;
  230.                 //それ以外の場合、(自分、親、親の親)の位置関係が大事
  231.                 }else if(this->state() == this->parent->state()){
  232.                     //GrandParentからの道が(右、右)or (左、左)の場合
  233.                     //GrandParentを一つ下に下ろして、自分を上げる
  234.                     this->parent->rotate();
  235.                     this->rotate();
  237.                 }else{
  238.                     //それ以外の場合、単純に自分を2回上に上げれば良い
  239.                     this->rotate();
  240.                     this->rotate();
  241.                 }
  242.             }
  246.             return;
  247.         }
  254.         /*
  255.             サイズなど、Nodeの持っている情報をupdateする(順番大事)
  256.             evaluate = trueなら遅延評価を下に伝播
  257.             
  258.         */
  259.         void update(bool evaluate = true){
  260.             if(isExist(this) == false)return;
  261.             if(evaluate)this->eval();
  262.             if(isExist(this->left) && evaluate)this->left->eval();
  263.             if(isExist(this->right) && evaluate)this->right->eval();
  266.             this->SubTreeSize = 1;
  267.             this->Min = this->Value;
  268.             this->Max = this->Value;
  269.             this->Sum = this->Value;
  272.             if(isExist(this->left)){
  273.                 this->SubTreeSize += this->left->SubTreeSize;
  274.                 if(this->left->Min < this->Min)this->Min = this->left->Min;
  275.                 if(this->left->Max > this->Max)this->Max = this->left->Max;
  276.                 this->Sum += this->left->Sum;
  277.             }
  279.             if(isExist(this->right)){
  280.                 this->SubTreeSize += this->right->SubTreeSize;
  281.                 if(this->right->Min < this->Min)this->Min = this->right->Min;
  282.                 if(this->right->Max > this->Max)this->Max = this->right->Max;
  283.                 this->Sum += this->right->Sum;
  285.             }
  287.             
  289.             
  291.             return;
  292.         }
  294.         /*
  295.             この頂点にアクセスされるたびに、各flagがtrueなら遅延させていた処理を実行し、
  296.             左右の子にflagを伝播する
  297.             各処理の概要
  298.             reverse : 自分の左右の子を入れ替えて、左右の子に遅延伝播(最終的に区間全体が反転する)
  299.             update : 自分自身のValue,Min,Maxをlazy_updateに変更し、SumにSubTreeSize*lazy_updateに変更
  300.             add : 自分自身のValue,Min,Maxにlazy_addを加算し、SumにSubTreeSize*lazy_addを加算
  302.             特に、回転を行う際は伝播の仕方に要注意
  304.             また、updateクエリが来た際は、それ以前の(add , update)クエリは全て無効になるので、
  305.             {
  306.                 lazy_add = 0;
  307.                 lazy_update = 更新する値
  308.             }
  309.             とし、updateクエリを左右の子に伝播させる時も、左右の子のlazy_addを0にして、lazy_updateを更新する
  311.             クエリの処理の順番は、
  312.             update -> addの順番で行う(上記の理由)
  314.             なお、rotateでのMin,Max,valueなどの更新に対応するため、
  315.             頂点xにアクセスし、左右に伝播させた時点でx->left,x->rightの持つ値を更新しておく
  317.         */
  318.         void eval(){
  321.             if(this->reverse_flag_lazy){
  322.                 swap(this->left , this->right);
  323.                 //下に伝播
  324.                 if(isExist(this->left))this->left->reverse_flag_lazy = !(this->left->reverse_flag_lazy);
  325.                 if(isExist(this->right))this->right->reverse_flag_lazy = !(this->right->reverse_flag_lazy);
  326.                 this->reverse_flag_lazy = false;
  327.             }
  329.             SplayNode *Left = this->left;
  330.             SplayNode *Right = this->right;
  332.             //updateクエリの処理が先
  333.             if(this->update_flag_lazy){
  334.                 this->Value = this->lazy_update;
  335.                 this->Min = this->lazy_update;
  336.                 this->Max = this->lazy_update;
  337.                 this->Sum = ( this->lazy_update )* (this->SubTreeSize);
  338.                 if(isExist(Left)){
  339.                     Left->update_flag_lazy = true;
  340.                     Left->lazy_update = this->lazy_update;
  341.                     Left->add_flag_lazy = false;//これまでの加算クエリは無駄になるから
  342.                     Left->lazy_add = 0;
  343.                 }
  344.                 if(isExist(Right)){
  345.                     Right->update_flag_lazy = true;
  346.                     Right->lazy_update = this->lazy_update;
  347.                     Right->add_flag_lazy = false;//これまでの加算クエリは無駄になるから
  348.                     Right->lazy_add = 0;
  349.                 }
  350.                 //伝播したらフラグを折る
  351.                 this->lazy_update = 0;
  352.                 this->update_flag_lazy = false;
  353.             }
  355.             if(this->add_flag_lazy){
  356.                 this->Value += this->lazy_add;
  357.                 this->Min += this->lazy_add;
  358.                 this->Max += this->lazy_add;
  359.                 this->Sum += ( this->lazy_add )* (this->SubTreeSize);
  360.                 if(isExist(Left)){
  361.                     Left->add_flag_lazy = true;//これまでの加算クエリは無駄になるから
  362.                     Left->lazy_add += this->lazy_add;
  363.                 }
  364.                 if(isExist(Right)){
  365.                     Right->add_flag_lazy = true;//これまでの加算クエリは無駄になるから
  366.                     Right->lazy_add += this->lazy_add;
  367.                 }
  368.                 //伝播したらフラグを折る
  369.                 this->lazy_add = 0;
  370.                 this->add_flag_lazy = false;
  371.             }
  372.         }

  535.         root = getNode(leftnum , root);
  536.         
  537.         SplayNode *leftRoot , *rightRoot;
  539.         leftRoot = root->left;
  540.         
  541.         rightRoot = root;

  545.

  572.         if(Size<index || index<0)return root;
  574.         std::pair<SplayNode*,SplayNode*> Trees = split(index,root);
  576.         SplayNode *leftRoot = Trees.first;
  577.         SplayNode *rightRoot = Trees.second;
  578.         Size++;
  580.         return merge(merge(leftRoot , NODE),rightRoot);
  581.     }
  591.     /*
  592.         rootを根とする部分木において、
  593.         左からindexの場所を削除して、両端をくっつける
  594.         firstは消したやつ以外をmergeした時のroot
  595.         secondは消したSplayNode
  596.     */
  597.     std::pair<SplayNode*,SplayNode*> Delete_sub(int index , SplayNode *root){
  598.         
  599.         if(isExist(root) == false){
  600.             
  601.             return std::make_pair(root,root);
  602.         }
  603.         if(index<0 || index >= Size){
  605.             return std::make_pair(root,root);
  606.         }
  607.         
  608.         if(Size == 1){
  610.             Size = 0;
  611.             root = nullptr;
  612.             return std::make_pair(nullptr,nullptr);
  613.         }
  615.         root = getNode(index,root);
  617.         SplayNode *leftRoot = root->left;
  618.         SplayNode *rightRoot = root->right;
  620.         if(isExist(leftRoot))leftRoot->parent = nullptr;
  621.         if(isExist(rightRoot))rightRoot->parent = nullptr;
  622.         root->left = nullptr;
  623.         root->right = nullptr;
  625.         Size--;
  626.         
  627.         root->update();
  628.         
  629.         return std::make_pair(merge(leftRoot,rightRoot) , root );
  630.     }

  653.     }

  704.     std::pair<SplayNode*,T> RangeQuery_sub(int l , int r , int mode , SplayNode *root){
